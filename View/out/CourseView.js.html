/**
 * Retrieves the details of a course based on the provided courseId.
 * 
 * @async
 * @function getCourseDetails
 * @param {Object} req - The request object from the client.
 * @param {Object} res - The response object to send data or errors back to the client.
 * @returns {Promise<void>} A promise that resolves with the appropriate HTTP response status and message.
 * 
 * @example
 * // Example usage:
 * // GET /courses/:courseId
 * app.get('/courses/:courseId', getCourseDetails);
 * 
 * @throws {Error} If the courseId is missing or the course cannot be found, an error response is returned.
 */
const getCourseDetails = async (req, res) => {
  const courseId = req.params.courseId;

  if (!courseId) {
      return res.status(400).json({
          message: "Bad Request: Missing courseId parameter"
      });
  }

  const course = await findCourseById(courseId); 

  if (!course) {
      return res.status(404).json({
          message: "Course not found"
      });
  }
  return res.status(200).json({
    message: "Course details retrieved successfully"
  });
};

/**
 * Searches for a course by its ID from the predefined course list.
 * 
 * @async
 * @function findCourseById
 * @param {string} courseId - The ID of the course to search for.
 * @returns {Object|null} The course object if found, or null if not found.
 * 
 * @example
 * // Example usage:
 * // const course = await findCourseById('course123');
 * 
 * @throws {Error} If any unexpected error occurs while fetching the course details.
 */
const findCourseById = async (courseId) => {
  const courses = [
    { id: "course123", name: "Advanced Programming" },
    { id: "course456", name: "Introduction to Algorithms" }
  ];

  return courses.find(course => course.id === courseId) || null;
};
